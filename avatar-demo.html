Integration Code Examples

1. Simple Integration

import { AvatarSelectionTrigger } from "@/components/avatars";

function MyComponent() {
  const [avatar, setAvatar] = useState(null);
  
  return (
    <AvatarSelectionTrigger
      onAvatarSelect={setAvatar}
      selectedAvatar={avatar}
      modalTitle="Choose Your Avatar"
    />
  );
}

2. Form Integration

function ProfileForm() {
  const [formData, setFormData] = useState({
    name: '',
    email: '',
    avatar: null
  });
  
  return (
    <form>
      <input name="name" ... />
      <input name="email" ... />
      
      <div className="form-group">
        <label>Profile Avatar:</label>
        <AvatarSelectionTrigger
          onAvatarSelect={(avatar) => 
            setFormData(prev => ({ ...prev, avatar }))
          }
          selectedAvatar={formData.avatar}
          className="w-full"
        />
      </div>
      
      <button type="submit">Save Profile</button>
    </form>
  );
}

3. Custom Workflow

function VideoCreationWorkflow() {
  const [modalOpen, setModalOpen] = useState(false);
  const [selectedAvatar, setSelectedAvatar] = useState(null);
  
  const handleAvatarSelect = (avatar) => {
    setSelectedAvatar(avatar);
    // Custom logic: validate avatar, check permissions, etc.
    if (avatar.premium && !userHasPremium) {
      showUpgradeModal();
      return;
    }
    // Proceed to next step
    proceedToScriptEditor(avatar);
  };
  
  return (
    <AvatarSelectionModal
      isOpen={modalOpen}
      onClose={() => setModalOpen(false)}
      onAvatarSelect={handleAvatarSelect}
      title="Select Video Presenter"
    />
  );
}
